# 数组去重

unique(arr, key) {
  const res = new Map();
  return arr.filter((a) => !res.has(a[key]) && res.set(a[key], 1))
}

# 按对象某个字段排序

compare (arr, key) {
  return arr.sort((a, b) => {return a[key] > b[key] ? 1 : a[key] < b[key] ? -1 : 0})
}

# 数组转为树

function arrayToTree(arr) {
  // 定义目标数组
  let obj = {}
  // 循环原始数据
  for (let i in arr) {
    if (arr[i].pid === "0") {
      obj = arr[i]
    } else {
      let item = arr.find(element => element.id === arr[i].pid)
      // if (item)
      item.children.push(arr[i])
    }
  }
  return obj
}
